  <script>
    (function () {
      const cameraInput   = document.getElementById('cameraInput');
      const galleryInput  = document.getElementById('galleryInput');
      const btnCamera     = document.getElementById('btnCamera');
      const btnGallery    = document.getElementById('btnGallery');
      const previewWrap   = document.getElementById('previewWrap');
      const previewImg    = document.getElementById('previewImg');
      const clearPreview  = document.getElementById('clearPreview');
      const submitBtn     = document.getElementById('submitBtn');
      const submitText    = submitBtn ? submitBtn.querySelector('.submit-text') : null;
      const form          = document.getElementById('uploadForm');
      const loadingMask   = document.getElementById('loadingMask');

      let activeInput = null;

      function setPreview(file) {
        if (!file) return;
        const reader = new FileReader();
        reader.onload = (e) => {
          previewImg.src = e.target.result;
          previewWrap.hidden = false;
          submitBtn.disabled = false;
          submitText.textContent = '开始识别';
        };
        reader.readAsDataURL(file);
      }

      function pickFile(inputEl) {
        // 把另外一个 input 清空，避免提交时双 input 冲突
        const other = inputEl === cameraInput ? galleryInput : cameraInput;
        other.value = '';
        activeInput = inputEl;
        inputEl.click();
      }

      btnCamera  && btnCamera.addEventListener('click',  () => pickFile(cameraInput));
      btnGallery && btnGallery.addEventListener('click', () => pickFile(galleryInput));

      [cameraInput, galleryInput].forEach((input) => {
        input && input.addEventListener('change', (e) => {
          const file = e.target.files && e.target.files[0];
          if (file) setPreview(file);
        });
      });

      clearPreview && clearPreview.addEventListener('click', () => {
        cameraInput.value = '';
        galleryInput.value = '';
        previewImg.src = '';
        previewWrap.hidden = true;
        submitBtn.disabled = true;
        submitText.textContent = '请选择图片';
      });

      form && form.addEventListener('submit', (e) => {
        // 必须有文件才提交
        const hasFile = (cameraInput.files && cameraInput.files[0]) ||
                        (galleryInput.files && galleryInput.files[0]);
        if (!hasFile) {
          e.preventDefault();
          return;
        }
        loadingMask.hidden = false;
        submitBtn.disabled = true;
        submitText.textContent = '识别中…';
      });
    })();
  </script>
